015package grails.plugin.springsecurity.annotation;
016
017import java.io.File;
018import java.io.FileReader;
019import java.io.IOException;
020import java.util.ArrayList;
021import java.util.List;
022import java.util.Properties;
023
024import org.codehaus.groovy.ast.ASTNode;
025import org.codehaus.groovy.ast.AnnotatedNode;
026import org.codehaus.groovy.ast.AnnotationNode;
027import org.codehaus.groovy.ast.ClassNode;
028import org.codehaus.groovy.ast.expr.ConstantExpression;
029import org.codehaus.groovy.ast.expr.Expression;
030import org.codehaus.groovy.ast.expr.ListExpression;
031import org.codehaus.groovy.control.CompilePhase;
032import org.codehaus.groovy.control.SourceUnit;
033import org.codehaus.groovy.control.messages.SyntaxErrorMessage;
034import org.codehaus.groovy.syntax.SyntaxException;
035import org.codehaus.groovy.transform.ASTTransformation;
036import org.codehaus.groovy.transform.GroovyASTTransformation;
037import org.springframework.util.StringUtils;
038
039/**
040 * See http://burtbeckwith.com/blog/?p=1398 for the motivation for this.
041 *
042 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
043 */
044@GroovyASTTransformation(phase=CompilePhase.CANONICALIZATION)
045public class AuthoritiesTransformation implements ASTTransformation {
046
047        protected static final ClassNode SECURED = new ClassNode(Secured.class);
048
049        public void visit(final ASTNode[] astNodes, final SourceUnit sourceUnit) {
050                ASTNode firstNode = astNodes[0];
051                ASTNode secondNode = astNodes[1];
052                try {
053                        if (!(firstNode instanceof AnnotationNode) || !(secondNode instanceof AnnotatedNode)) {
054                                throw new IllegalArgumentException("Internal error: wrong types: " + firstNode.getClass().getName() +
055                                                " / " + secondNode.getClass().getName());
056                        }
057
058                        AnnotationNode rolesAnnotationNode = (AnnotationNode) firstNode;
059                        AnnotatedNode annotatedNode = (AnnotatedNode) secondNode;
060
061                        AnnotationNode secured = createAnnotation(rolesAnnotationNode, sourceUnit);
062                        if (secured != null) {
063                                annotatedNode.addAnnotation(secured);
064                        }
065                }
066                catch (Exception e) {
067                        reportError(e.getMessage(), sourceUnit, firstNode);
068                }
069        }
070
071        protected AnnotationNode createAnnotation(final AnnotationNode rolesNode, final SourceUnit sourceUnit) throws IOException {
072                Expression value = rolesNode.getMembers().get("value");
073                if (!(value instanceof ConstantExpression)) {
074                        reportError("annotation @Authorities value isn't a ConstantExpression: " + value, sourceUnit, rolesNode);
075                        return null;
076                }
077
078                String fieldName = value.getText();
079                String[] authorityNames = getAuthorityNames(fieldName, rolesNode, sourceUnit);
080                if (authorityNames == null) {
081                        return null;
082                }
083
084                return buildAnnotationNode(authorityNames);
085        }
086
087        protected AnnotationNode buildAnnotationNode(String[] authorityNames) {
088                AnnotationNode securedAnnotationNode = new AnnotationNode(SECURED);
089                List<Expression> nameExpressions = new ArrayList<Expression>();
090                for (String authorityName : authorityNames) {
091                        nameExpressions.add(new ConstantExpression(authorityName));
092                }
093                securedAnnotationNode.addMember("value", new ListExpression(nameExpressions));
094                return securedAnnotationNode;
095        }
096
097        protected String[] getAuthorityNames(final String fieldName, final AnnotationNode rolesNode, final SourceUnit sourceUnit) throws IOException {
098
099                Properties properties = new Properties();
100                File propertyFile = new File("roles.properties");
101                if (!propertyFile.exists()) {
102                        reportError("Property file roles.properties not found", sourceUnit, rolesNode);
103                        return null;
104                }
105
106                properties.load(new FileReader(propertyFile));
107
108                Object value = properties.getProperty(fieldName);
109                if (value == null) {
110                        reportError("No value for property '" + fieldName + "'", sourceUnit, rolesNode);
111                        return null;
112                }
113
114                List<String> names = new ArrayList<String>();
115                for (String auth : StringUtils.commaDelimitedListToStringArray(value.toString())) {
116                        auth = auth.trim();
117                        if (auth.length() > 0) {
118                                names.add(auth);
119                        }
120                }
121
122                return names.toArray(new String[names.size()]);
123        }
124
125        protected void reportError(final String message, final SourceUnit sourceUnit, final ASTNode node) {
126                SyntaxException se = new SyntaxException(message, node.getLineNumber(), node.getColumnNumber());
127                sourceUnit.getErrorCollector().addErrorAndContinue(new SyntaxErrorMessage(se, sourceUnit));
128        }
129}
