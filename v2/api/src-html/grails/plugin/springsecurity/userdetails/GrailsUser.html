015package grails.plugin.springsecurity.userdetails;
016
017import java.util.Collection;
018
019import org.springframework.security.core.GrantedAuthority;
020import org.springframework.security.core.userdetails.User;
021
022/**
023 * Extends the default Spring Security user class to contain the ID for efficient lookup
024 * of the domain class from the Authentication.
025 *
026 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
027 */
028public class GrailsUser extends User {
029
030        private static final long serialVersionUID = 1;
031
032        private final Object id;
033
034        /**
035         * Constructor.
036         *
037         * @param username the username presented to the
038         *        <code>DaoAuthenticationProvider</code>
039         * @param password the password that should be presented to the
040         *        <code>DaoAuthenticationProvider</code>
041         * @param enabled set to <code>true</code> if the user is enabled
042         * @param accountNonExpired set to <code>true</code> if the account has not expired
043         * @param credentialsNonExpired set to <code>true</code> if the credentials have not expired
044         * @param accountNonLocked set to <code>true</code> if the account is not locked
045         * @param authorities the authorities that should be granted to the caller if they
046         *        presented the correct username and password and the user is enabled. Not null.
047         * @param id the id of the domain class instance used to populate this
048         */
049        public GrailsUser(String username, String password, boolean enabled, boolean accountNonExpired,
050         boolean credentialsNonExpired, boolean accountNonLocked,
051         Collection<GrantedAuthority> authorities, Object id) {
052                super(username, password, enabled, accountNonExpired, credentialsNonExpired,
053                                accountNonLocked, authorities);
054                this.id = id;
055        }
056
057        /**
058         * Get the id.
059         * @return the id
060         */
061        public Object getId() {
062                return id;
063        }
064}
