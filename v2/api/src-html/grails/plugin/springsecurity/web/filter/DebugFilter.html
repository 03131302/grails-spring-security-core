015package grails.plugin.springsecurity.web.filter;
016
017import grails.util.GrailsUtil;
018
019import java.io.IOException;
020import java.io.PrintWriter;
021import java.io.StringWriter;
022import java.util.List;
023
024import javax.servlet.Filter;
025import javax.servlet.FilterChain;
026import javax.servlet.ServletException;
027import javax.servlet.ServletRequest;
028import javax.servlet.ServletResponse;
029import javax.servlet.http.HttpServletRequest;
030import javax.servlet.http.HttpServletRequestWrapper;
031import javax.servlet.http.HttpServletResponse;
032import javax.servlet.http.HttpSession;
033
034import org.slf4j.Logger;
035import org.slf4j.LoggerFactory;
036import org.springframework.security.web.FilterChainProxy;
037import org.springframework.security.web.SecurityFilterChain;
038import org.springframework.security.web.util.UrlUtils;
039import org.springframework.web.filter.GenericFilterBean;
040
041/**
042 * Based on the package-scope org.springframework.security.config.debug.DebugFilter.
043 *
044 * @author Luke Taylor
045 * @author Rob Winch
046 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
047 */
048public class DebugFilter extends GenericFilterBean {
049
050        protected static final String ALREADY_FILTERED_ATTR_NAME = DebugFilter.class.getName() + ".FILTERED";
051
052        protected final FilterChainProxy filterChainProxy;
053        protected final Logger log = LoggerFactory.getLogger(getClass());
054
055        public DebugFilter(FilterChainProxy fcp) {
056                filterChainProxy = fcp;
057        }
058
059        public void doFilter(ServletRequest req, ServletResponse res, FilterChain filterChain) throws ServletException, IOException {
060
061                HttpServletRequest request = (HttpServletRequest)req;
062                HttpServletResponse response = (HttpServletResponse)res;
063
064                List<Filter> filters = getFilters(request);
065                log(false, "Request received for '{}':\n\n{}\n\nservletPath:{}\npathInfo:{}\n\n{}",
066                                UrlUtils.buildRequestUrl(request), request, request.getServletPath(), request.getPathInfo(), formatFilters(filters));
067
068                if (request.getAttribute(ALREADY_FILTERED_ATTR_NAME) == null) {
069                        invokeWithWrappedRequest(request, response, filterChain);
070                }
071                else {
072                        filterChainProxy.doFilter(request, response, filterChain);
073                }
074        }
075
076        protected void invokeWithWrappedRequest(HttpServletRequest request, HttpServletResponse response,
077                        FilterChain filterChain) throws IOException, ServletException {
078
079                request.setAttribute(ALREADY_FILTERED_ATTR_NAME, true);
080
081                request = new HttpServletRequestWrapper(request) {
082
083                        @Override
084                        public HttpSession getSession() {
085                                boolean sessionExists = super.getSession(false) != null;
086                                HttpSession session = super.getSession();
087
088                                if (!sessionExists) {
089                                        log(true, "New HTTP session created: {}", session.getId());
090                                }
091
092                                return session;
093                        }
094
095                        @Override
096                        public HttpSession getSession(boolean create) {
097                                return create ? getSession() : super.getSession(false);
098                        }
099                };
100
101                try {
102                        filterChainProxy.doFilter(request, response, filterChain);
103                }
104                finally {
105                        request.removeAttribute(ALREADY_FILTERED_ATTR_NAME);
106                }
107        }
108
109        protected String formatFilters(List<Filter> filters) {
110                StringBuilder sb = new StringBuilder("Security filter chain: ");
111                if (filters == null) {
112                        sb.append("no match");
113                }
114                else if (filters.isEmpty()) {
115                        sb.append("[] empty (bypassed by security='none') ");
116                }
117                else {
118                        sb.append("[\n");
119                        for (Filter f : filters) {
120                                sb.append("  ").append(f.getClass().getSimpleName()).append("\n");
121                        }
122                        sb.append("]");
123                }
124
125                return sb.toString();
126        }
127
128        protected List<Filter> getFilters(HttpServletRequest request) {
129                for (SecurityFilterChain chain : filterChainProxy.getFilterChains()) {
130                        if (chain.matches(request)) {
131                                return chain.getFilters();
132                        }
133                }
134
135                return null;
136        }
137
138        protected void log(boolean dumpStack, String message, Object... args) {
139                StringBuilder output = new StringBuilder(256);
140                output.append("\n\n************************************************************\n\n");
141                output.append(message).append("\n");
142
143                if (dumpStack) {
144                        StringWriter os = new StringWriter();
145                        GrailsUtil.deepSanitize(new Exception()).printStackTrace(new PrintWriter(os));
146                        StringBuffer buffer = os.getBuffer();
147                        // Remove the exception in case it scares people.
148                        int start = buffer.indexOf("java.lang.Exception");
149                        buffer.replace(start, start + 19, "");
150                        output.append("\nCall stack: \n").append(os);
151                }
152
153                output.append("\n\n************************************************************\n\n");
154                log.info(output.toString(), args);
155        }
156
157        public FilterChainProxy getFilterChainProxy() {
158                return filterChainProxy;
159        }
160}
