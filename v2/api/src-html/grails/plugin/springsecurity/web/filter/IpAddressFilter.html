015package grails.plugin.springsecurity.web.filter;
016
017import grails.plugin.springsecurity.InterceptedUrl;
018import grails.plugin.springsecurity.ReflectionUtils;
019
020import java.io.IOException;
021import java.util.ArrayList;
022import java.util.List;
023import java.util.Map;
024
025import javax.servlet.FilterChain;
026import javax.servlet.ServletException;
027import javax.servlet.ServletRequest;
028import javax.servlet.ServletResponse;
029import javax.servlet.http.HttpServletRequest;
030import javax.servlet.http.HttpServletResponse;
031
032import org.codehaus.groovy.grails.web.util.WebUtils;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035import org.springframework.security.access.ConfigAttribute;
036import org.springframework.security.web.util.matcher.IpAddressMatcher;
037import org.springframework.util.AntPathMatcher;
038import org.springframework.util.Assert;
039import org.springframework.util.StringUtils;
040import org.springframework.web.filter.GenericFilterBean;
041
042/**
043 * Blocks access to protected resources based on IP address. Sends 404 rather than
044 * reporting error to hide visibility of the resources.
045 * <br/>
046 * Supports either single IP addresses or CIDR masked patterns
047 * (e.g. 192.168.1.0/24, 202.24.0.0/14, 10.0.0.0/8, etc.).
048 *
049 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
050 */
051public class IpAddressFilter extends GenericFilterBean {
052
053        protected static final String IPV4_LOOPBACK = "127.0.0.1";
054        protected static final String IPV6_LOOPBACK = "0:0:0:0:0:0:0:1";
055
056        protected final Logger log = LoggerFactory.getLogger(getClass());
057
058        protected final AntPathMatcher pathMatcher = new AntPathMatcher();
059
060        protected List<InterceptedUrl> restrictions;
061        protected boolean allowLocalhost = true;
062
063        public void doFilter(final ServletRequest req, final ServletResponse res, final FilterChain chain)
064                                throws IOException, ServletException {
065
066                HttpServletRequest request = (HttpServletRequest)req;
067                HttpServletResponse response = (HttpServletResponse)res;
068
069                if (!isAllowed(request)) {
070                        deny(request, response);
071                        return;
072                }
073
074                chain.doFilter(request, response);
075        }
076
077        protected void deny(final HttpServletRequest req, final HttpServletResponse res) throws IOException {
078                // send 404 to hide the existence of the resource
079                res.sendError(HttpServletResponse.SC_NOT_FOUND);
080        }
081
082        @Override
083        protected void initFilterBean() {
084                Assert.notNull(restrictions, "ipRestrictions map is required");
085        }
086
087        /**
088         * Dependency injection for the ip/pattern restriction map. Keys are URL patterns and values
089         * are either single <code>String</code>s or <code>List</code>s of <code>String</code>s
090         * representing IP address patterns to allow for the specified URLs.
091         *
092         * @param ipRestrictions the map
093         */
094        public void setIpRestrictions(final Map<String, Object> ipRestrictions) {
095                restrictions = ReflectionUtils.splitMap(ipRestrictions, false);
096        }
097
098        /**
099         * Dependency injection for whether to allow localhost calls (useful for testing).
100         * TODO document
101         *
102         * @param allow if <code>true</code> allow localhost access
103         */
104        public void setAllowLocalhost(boolean allow) {
105                allowLocalhost = allow;
106        }
107
108        protected boolean isAllowed(final HttpServletRequest request) {
109                String ip = request.getRemoteAddr();
110                if (allowLocalhost && (IPV4_LOOPBACK.equals(ip) || IPV6_LOOPBACK.equals(ip))) {
111                        return true;
112                }
113
114                String uri = (String)request.getAttribute(WebUtils.FORWARD_REQUEST_URI_ATTRIBUTE);
115                if (!StringUtils.hasLength(uri)) {
116                        uri = request.getRequestURI();
117                        if (!request.getContextPath().equals("/") && uri.startsWith(request.getContextPath())) {
118                                uri = uri.substring(request.getContextPath().length());
119                        }
120                }
121
122                List<InterceptedUrl> matching = findMatchingRules(uri);
123                if (matching.isEmpty()) {
124                        return true;
125                }
126
127                for (InterceptedUrl iu : matching) {
128                        for (ConfigAttribute ipPattern : iu.getConfigAttributes()) {
129                                if (new IpAddressMatcher(ipPattern.getAttribute()).matches(request)) {
130                                        return true;
131                                }
132                        }
133                }
134
135                log.warn("disallowed request {} from {}", new Object[] { uri, ip });
136                return false;
137        }
138
139        protected List<InterceptedUrl> findMatchingRules(String uri) {
140                List<InterceptedUrl> matching = new ArrayList<InterceptedUrl>();
141                for (InterceptedUrl iu : restrictions) {
142                        if (pathMatcher.match(iu.getPattern(), uri)) {
143                                matching.add(iu);
144                        }
145                }
146                return matching;
147        }
148}
