001package grails.plugin.springsecurity.annotation;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Inherited;
006import java.lang.annotation.Retention;
007import java.lang.annotation.RetentionPolicy;
008import java.lang.annotation.Target;
009
010import org.codehaus.groovy.transform.GroovyASTTransformationClass;
011
012/**
013 * Specify the property file key with this annotation, and the AST transform
014 * class will replace with an @Secured annotation with the associated role names.
015 *
016 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
017 */
018@Target({ElementType.FIELD, ElementType.METHOD, ElementType.TYPE})
019@Retention(RetentionPolicy.RUNTIME)
020@Inherited
021@Documented
022@GroovyASTTransformationClass("grails.plugin.springsecurity.annotation.AuthoritiesTransformation")
023public @interface Authorities {
024        /**
025         * The property file key; the property value will be a comma-delimited list of role names.
026         * @return the key
027         */
028        String value();
029}
