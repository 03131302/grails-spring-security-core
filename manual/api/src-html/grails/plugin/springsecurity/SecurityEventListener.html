015package grails.plugin.springsecurity;
016
017import groovy.lang.Closure;
018
019import org.springframework.context.ApplicationContext;
020import org.springframework.context.ApplicationContextAware;
021import org.springframework.context.ApplicationEvent;
022import org.springframework.context.ApplicationListener;
023import org.springframework.security.access.event.AbstractAuthorizationEvent;
024import org.springframework.security.authentication.event.AbstractAuthenticationEvent;
025import org.springframework.security.authentication.event.AbstractAuthenticationFailureEvent;
026import org.springframework.security.authentication.event.AuthenticationSuccessEvent;
027import org.springframework.security.authentication.event.InteractiveAuthenticationSuccessEvent;
028import org.springframework.security.web.authentication.switchuser.AuthenticationSwitchUserEvent;
029
030/**
031 * Registers as an event listener and delegates handling of security-related events
032 * to optional closures defined in Config.groovy.
033 * <p/>
034 * The following callbacks are supported:<br/>
035 * <ul>
036 * <li>onInteractiveAuthenticationSuccessEvent</li>
037 * <li>onAbstractAuthenticationFailureEvent</li>
038 * <li>onAuthenticationSuccessEvent</li>
039 * <li>onAuthenticationSwitchUserEvent</li>
040 * <li>onAuthorizationEvent</li>
041 * </ul>
042 * All callbacks are optional; you can implement just the ones you're interested in, e.g.
043 * <pre>
044 * grails {
045 *    plugin {
046 *       springsecurity {
047 *          ...
048 *          onAuthenticationSuccessEvent = { e, appCtx ->
049 *             ...
050 *          }
051 *       }
052 *    }
053 * }
054 * </pre>
055 * The event and the Spring context are provided in case you need to look up a Spring bean,
056 * e.g. the Hibernate SessionFactory.
057 *
058 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
059 */
060public class SecurityEventListener implements ApplicationListener<ApplicationEvent>, ApplicationContextAware {
061
062        protected ApplicationContext applicationContext;
063
064        /**
065         * {@inheritDoc}
066         * @see org.springframework.context.ApplicationListener#onApplicationEvent(
067         *      org.springframework.context.ApplicationEvent)
068         */
069        public void onApplicationEvent(final ApplicationEvent e) {
070                if (e instanceof AbstractAuthenticationEvent) {
071                        if (e instanceof InteractiveAuthenticationSuccessEvent) {
072                                call(e, "onInteractiveAuthenticationSuccessEvent");
073                        }
074                        else if (e instanceof AbstractAuthenticationFailureEvent) {
075                                call(e, "onAbstractAuthenticationFailureEvent");
076                        }
077                        else if (e instanceof AuthenticationSuccessEvent) {
078                                call(e, "onAuthenticationSuccessEvent");
079                        }
080                        else if (e instanceof AuthenticationSwitchUserEvent) {
081//                              GrailsUser userInfo = (GrailsUser)event.getAuthentication().getPrincipal()
082//                              UserDetails userDetails = event.getTargetUser()
083                                call(e, "onAuthenticationSwitchUserEvent");
084                        }
085                }
086                else if (e instanceof AbstractAuthorizationEvent) {
087                        call(e, "onAuthorizationEvent");
088                }
089        }
090
091        @SuppressWarnings("rawtypes")
092        protected void call(final ApplicationEvent e, final String closureName) {
093                Object closure = SpringSecurityUtils.getSecurityConfig().get(closureName);
094                if (closure instanceof Closure) {
095                        ((Closure)closure).call(new Object[] { e, applicationContext });
096                }
097        }
098
099        /**
100         * {@inheritDoc}
101         * @see org.springframework.context.ApplicationContextAware#setApplicationContext(
102         *      org.springframework.context.ApplicationContext)
103         */
104        public void setApplicationContext(final ApplicationContext ctx) {
105                applicationContext = ctx;
106        }
107}
