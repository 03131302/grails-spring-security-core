049public abstract class AbstractFilterInvocationDefinition implements FilterInvocationSecurityMetadataSource, InitializingBean {

051        protected static final Collection<ConfigAttribute> DENY;
052        static {
053                Collection<ConfigAttribute> list = new ArrayList<ConfigAttribute>(1);
054                list.add(new SecurityConfig("_DENY_"));
055                DENY = Collections.unmodifiableCollection(list);
056        }
057
058        protected boolean rejectIfNoRule;
059        protected RoleVoter roleVoter;
060        protected AuthenticatedVoter authenticatedVoter;
061        protected final List<InterceptedUrl> compiled = Collections.synchronizedList(new ArrayList<InterceptedUrl>());
062   protected MessageSourceAccessor messages = SpringSecurityMessageSource.getAccessor();
063        protected AntPathMatcher urlMatcher = new AntPathMatcher();
064        protected boolean initialized;
065        protected boolean grails23Plus;
066
067        protected final Logger log = LoggerFactory.getLogger(getClass());

074                // override if necessary
075        }

077        /**
078         * {@inheritDoc}
079         * @see org.springframework.security.access.SecurityMetadataSource#getAttributes(java.lang.Object)
080         */
081        public Collection<ConfigAttribute> getAttributes(Object object) throws IllegalArgumentException {
082                Assert.isTrue(object != null && supports(object.getClass()), "Object must be a FilterInvocation");
083
084                FilterInvocation filterInvocation = (FilterInvocation)object;
085
086                String url = determineUrl(filterInvocation);
087
088                Collection<ConfigAttribute> configAttributes;
089                try {
090                        configAttributes = findConfigAttributes(url, filterInvocation.getRequest().getMethod());
091                }
092                catch (RuntimeException e) {
093                        throw e;
094                }
095                catch (Exception e) {
096                        throw new RuntimeException(e);
097                }
098
099                if ((configAttributes == null || configAttributes.isEmpty()) && rejectIfNoRule) {
100                        // return something that cannot be valid; this will cause the voters to abstain or deny
101                        return DENY;
102                }
103
104                return configAttributes;
105        }
106
107        protected String determineUrl(final FilterInvocation filterInvocation) {
108                return lowercaseAndStripQuerystring(calculateUri(filterInvocation.getHttpRequest()));
109        }
110
111        protected boolean stopAtFirstMatch() {
112                return false;
113        }
114
115        // for testing
116        public InterceptedUrl getInterceptedUrl(final String url, final HttpMethod httpMethod) throws Exception {
117
118                initialize();
119
120                for (InterceptedUrl iu : compiled) {
121                        if (iu.getHttpMethod() == httpMethod && iu.getPattern().equals(url)) {
122                                return iu;
123                        }
124                }
125
126                return null;
127        }
128
129        protected Collection<ConfigAttribute> findConfigAttributes(final String url, final String requestMethod) throws Exception {
130
131                initialize();
132
133                Collection<ConfigAttribute> configAttributes = null;
134                String configAttributePattern = null;
135
136                boolean stopAtFirstMatch = stopAtFirstMatch();
137                for (InterceptedUrl iu : compiled) {
138
139                        if (iu.getHttpMethod() != null && requestMethod != null && iu.getHttpMethod() != HttpMethod.valueOf(requestMethod)) {
140                                if (log.isDebugEnabled()) {
141                                        log.debug("Request '{} {}' doesn't match '{} {}'", new Object[] { requestMethod, url, iu.getHttpMethod(), iu.getPattern() });
142                                }
143                                continue;
144                        }
145
146                        if (urlMatcher.match(iu.getPattern(), url)) {
147                                if (configAttributes == null || urlMatcher.match(configAttributePattern, iu.getPattern())) {
148                                        configAttributes = iu.getConfigAttributes();
149                                        configAttributePattern = iu.getPattern();
150                                        if (log.isTraceEnabled()) {
151                                                log.trace("new candidate for '{}': '{}':{}", new Object[] { url, iu.getPattern(), configAttributes });
152                                        }
153                                        if (stopAtFirstMatch) {
154                                                break;
155                                        }
156                                }
157                        }
158                }
159
160                if (log.isTraceEnabled()) {
161                        if (configAttributes == null) {
162                                log.trace("no config for '{}'", url);
163                        }
164                        else {
165                                log.trace("config for '{}' is '{}':{}", new Object[] { url, configAttributePattern, configAttributes });
166                        }
167                }
168
169                return configAttributes;
170        }

172        protected void initialize() throws Exception {
173                // override if necessary
174        }

176        /**
177         * {@inheritDoc}
178         * @see org.springframework.security.access.SecurityMetadataSource#supports(java.lang.Class)

180        public boolean supports(Class<?> clazz) {
181                return FilterInvocation.class.isAssignableFrom(clazz);
182        }

184        /**
185         * {@inheritDoc}
186         * @see org.springframework.security.access.SecurityMetadataSource#getAllConfigAttributes()
187         */
188        public Collection<ConfigAttribute> getAllConfigAttributes() {
189                try {
190                        initialize();
191                }
192                catch (Exception e) {
193                        GrailsUtil.deepSanitize(e);
194                        log.error(e.getMessage(), e);
195                }
196
197                Collection<ConfigAttribute> all = new LinkedHashSet<ConfigAttribute>();
198                for (InterceptedUrl iu : compiled) {
199                        all.addAll(iu.getConfigAttributes());
200                }
201                return Collections.unmodifiableCollection(all);
202        }
203
204        protected String calculateUri(final HttpServletRequest request) {
205                String url = request.getRequestURI().substring(request.getContextPath().length());
206                int semicolonIndex = url.indexOf(";");
207                return semicolonIndex == -1 ? url : url.substring(0, semicolonIndex);
208        }
209
210        protected String lowercaseAndStripQuerystring(final String url) {
211
212                String fixed = url.toLowerCase();
213
214                int firstQuestionMarkIndex = fixed.indexOf("?");
215                if (firstQuestionMarkIndex != -1) {
216                        fixed = fixed.substring(0, firstQuestionMarkIndex);
217                }
218
219                return fixed;
220        }
221
222        protected AntPathMatcher getUrlMatcher() {
223                return urlMatcher;
224        }

227         * For debugging.
228         * @return an unmodifiable map of {@link AnnotationFilterInvocationDefinition}ConfigAttributeDefinition
229         * keyed by compiled patterns

231        public List<InterceptedUrl> getConfigAttributeMap() {
232                return Collections.unmodifiableList(compiled);
233        }

235        // fixes extra spaces, trailing commas, etc.
236        protected List<String> split(final String value) {
237                if (!value.startsWith("ROLE_") && !value.startsWith("IS_")) {
238                        // an expression
239                        return Collections.singletonList(value);
240                }
241
242                String[] parts = StringUtils.commaDelimitedListToStringArray(value);
243                List<String> cleaned = new ArrayList<String>();
244                for (String part : parts) {
245                        part = part.trim();
246                        if (part.length() > 0) {
247                                cleaned.add(part);
248                        }
249                }
250                return cleaned;
251        }
252
253        protected void compileAndStoreMapping(InterceptedUrl iu) {
254                String pattern = iu.getPattern();
255                HttpMethod method = iu.getHttpMethod();
256
257                String key = pattern.toLowerCase();
258
259                Collection<ConfigAttribute> configAttributes = iu.getConfigAttributes();
260
261                InterceptedUrl replaced = storeMapping(key, method, Collections.unmodifiableCollection(configAttributes));
262                if (replaced != null) {
263                        log.warn("replaced rule for '{}' with roles {} with roles {}", new Object[] { key, replaced.getConfigAttributes(), configAttributes });
264                }
265        }
266
267        protected InterceptedUrl storeMapping(final String pattern, final HttpMethod method,
268                        final Collection<ConfigAttribute> configAttributes) {
269
270                InterceptedUrl existing = null;
271                for (Iterator<InterceptedUrl> iter = compiled.iterator(); iter.hasNext(); ) {
272                        InterceptedUrl iu = iter.next();
273                        if (iu.getPattern().equals(pattern) && iu.getHttpMethod() == method) {
274                                existing = iu;
275                                iter.remove();
276                                break;
277                        }
278                }
279
280                compiled.add(new InterceptedUrl(pattern, method, configAttributes));
281
282                return existing;
283        }
284
285        protected void resetConfigs() {
286                compiled.clear();
287        }
288
289        /**
290         * For admin/debugging - find all config attributes that apply to the specified URL (doesn't consider request method restrictions).
291         * @param url the URL
292         * @return matching attributes
293         */
294        public Collection<ConfigAttribute> findMatchingAttributes(final String url) {
295                for (InterceptedUrl iu : compiled) {
296                        if (urlMatcher.match(iu.getPattern(), url)) {
297                                return iu.getConfigAttributes();
298                        }
299                }
300                return Collections.emptyList();
301        }

303        /**
304         * Dependency injection for whether to reject if there's no matching rule.
305         * @param reject if true, reject access unless there's a pattern for the specified resource
306         */
307        public void setRejectIfNoRule(final boolean reject) {
308                rejectIfNoRule = reject;
309        }

311        public void afterPropertiesSet() {
312                String version = Metadata.getCurrent().getGrailsVersion();
313                grails23Plus = !version.startsWith("2.0") && !version.startsWith("2.1") && !version.startsWith("2.2");
314        }
