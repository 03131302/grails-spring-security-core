015package grails.plugin.springsecurity.authentication.encoding;
016
017import java.security.MessageDigest;
018import java.security.NoSuchAlgorithmException;
019
020import org.springframework.beans.factory.InitializingBean;
021import org.springframework.security.crypto.codec.Hex;
022import org.springframework.util.Assert;
023
024/**
025 * Workaround for the limitation in Digest authentication where you must store passwords in
026 * the database in cleartext so the encrypted password use for comparison with what the
027 * browser sends will agree. This password encoder uses the same algorithm as the browser
028 * and does a good job with encryption (it effectively uses the username and the Realm
029 * name as a salt) but is completely unconfigurable.
030 * <p/>
031 * Uses code from package org.springframework.security.web.authentication.www.DigestAuthUtils
032 * which is unfortunately package-default.
033 *
034 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
035 */
036@SuppressWarnings("deprecation")
037public class DigestAuthPasswordEncoder implements org.springframework.security.authentication.encoding.PasswordEncoder, InitializingBean {
038
039        protected String realm;
040        protected boolean initializing = true;
041
042        public String encodePassword(final String rawPass, final Object salt) {
043                String username;
044                if (initializing && "userNotFoundPassword".equals(rawPass)) {
045                        // during startup, DaoAuthenticationProvider calls this method with a null salt
046                        username = rawPass;
047                }
048                else {
049                        Assert.notNull(salt, "Salt is required and must be the username");
050                        username = salt.toString();
051                }
052                return md5Hex(username + ":" + realm + ":" + rawPass);
053        }
054
055        public boolean isPasswordValid(final String encPass, final String rawPass, final Object salt) {
056                // the 'raw' password will already be encrypted, so compare directly
057                return encPass != null && rawPass != null ? rawPass.equals(encPass) : false;
058        }
059
060        /**
061         * Dependency injection for the realm name.
062         *
063         * @param name the name
064         */
065        public void setRealm(final String name) {
066                realm = name;
067        }
068
069        /**
070         * Called after the ApplicationContext is built to enable standard behavior.
071         */
072        public void resetInitializing() {
073           initializing = false;
074   }
075
076        public void afterPropertiesSet() {
077                Assert.hasLength(realm, "realm is required");
078        }
079
080        protected String md5Hex(final String s) {
081                MessageDigest digest;
082                try {
083                        digest = MessageDigest.getInstance("MD5");
084                }
085                catch (NoSuchAlgorithmException e) {
086                        throw new IllegalStateException("No MD5 algorithm available!");
087                }
088
089                return new String(Hex.encode(digest.digest(s.getBytes())));
090        }
091}
