015package grails.plugin.springsecurity.web.access.intercept;
016
017import java.util.Collection;
018import java.util.LinkedHashMap;
019import java.util.Map;
020
021import org.springframework.beans.factory.FactoryBean;
022import org.springframework.beans.factory.InitializingBean;
023import org.springframework.security.access.ConfigAttribute;
024import org.springframework.security.access.SecurityConfig;
025import org.springframework.security.web.access.intercept.DefaultFilterInvocationSecurityMetadataSource;
026import org.springframework.security.web.access.intercept.FilterInvocationSecurityMetadataSource;
027import org.springframework.security.web.util.matcher.AntPathRequestMatcher;
028import org.springframework.security.web.util.matcher.RequestMatcher;
029import org.springframework.util.AntPathMatcher;
030import org.springframework.util.Assert;
031
032/**
033 * Factory bean that builds a {@link FilterInvocationSecurityMetadataSource} for channel security.
034 *
035 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
036 */
037public class ChannelFilterInvocationSecurityMetadataSourceFactoryBean
038       implements FactoryBean<FilterInvocationSecurityMetadataSource>, InitializingBean {
039
040        protected AntPathMatcher urlMatcher = new AntPathMatcher();
041        protected Map<String, String> definition;
042        protected DefaultFilterInvocationSecurityMetadataSource source;
043
044        /**
045         * {@inheritDoc}
046         * @see org.springframework.beans.factory.FactoryBean#getObject()
047         */
048        public FilterInvocationSecurityMetadataSource getObject() {
049                return source;
050        }
051
052        /**
053         * {@inheritDoc}
054         * @see org.springframework.beans.factory.FactoryBean#getObjectType()
055         */
056        public Class<DefaultFilterInvocationSecurityMetadataSource> getObjectType() {
057                return DefaultFilterInvocationSecurityMetadataSource.class;
058        }
059
060        /**
061         * {@inheritDoc}
062         * @see org.springframework.beans.factory.FactoryBean#isSingleton()
063         */
064        public boolean isSingleton() {
065                return true;
066        }
067
068        /**
069         * {@inheritDoc}
070         * @see org.springframework.beans.factory.InitializingBean#afterPropertiesSet()
071         */
072        public void afterPropertiesSet() {
073                Assert.notNull(definition, "definition map is required");
074                Assert.notNull(urlMatcher, "urlMatcher is required");
075
076                source = new DefaultFilterInvocationSecurityMetadataSource(buildMap());
077        }
078
079        protected LinkedHashMap<RequestMatcher, Collection<ConfigAttribute>> buildMap() {
080                LinkedHashMap<RequestMatcher, Collection<ConfigAttribute>> map = new LinkedHashMap<RequestMatcher, Collection<ConfigAttribute>>();
081                for (Map.Entry<String, String> entry : definition.entrySet()) {
082                        String value = entry.getValue();
083                        if (value == null) {
084                                throw new IllegalArgumentException("The rule for URL '" + value + "' cannot be null");
085                        }
086                        value = value.trim();
087
088                        if (!"ANY_CHANNEL".equals(value) &&
089                                        !"REQUIRES_SECURE_CHANNEL".equals(value) &&
090                                        !"REQUIRES_INSECURE_CHANNEL".equals(value)) {
091                                throw new IllegalArgumentException("The rule for URL '" + value +
092                                                "' must be one of REQUIRES_SECURE_CHANNEL, REQUIRES_INSECURE_CHANNEL, or ANY_CHANNEL");
093                        }
094
095                        map.put(new AntPathRequestMatcher(entry.getKey()), SecurityConfig.createList(value));
096                }
097                return map;
098        }
099
100        /**
101         * Dependency injection for the definition map.
102         *
103         * @param d keys are URL patterns, values are ANY_CHANNEL, REQUIRES_SECURE_CHANNEL, or REQUIRES_INSECURE_CHANNEL
104         */
105        public void setDefinition(Map<String, String> d) {
106                definition = d;
107        }
108}
